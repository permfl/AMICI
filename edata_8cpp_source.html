61 ExpData::ExpData(ReturnData const& rdata, std::vector<realtype> sigma_y, std::vector<realtype> sigma_z)
62  : ExpData(rdata.nytrue, rdata.nztrue, rdata.nmaxevent, rdata.ts)
63 {
64  if (sigma_y.size() != (unsigned) nytrue && sigma_y.size() != (unsigned) nytrue*nt())
65  throw AmiException("Dimension of sigma_y must be %d or %d, was %d", nytrue, nytrue*nt(), sigma_y.size());
66 
67  if (sigma_z.size() != (unsigned) nztrue && sigma_z.size() != (unsigned) nztrue*nmaxevent)
68  throw AmiException("Dimension of sigma_z must be %d or %d, was %d", nztrue, nztrue*nmaxevent, sigma_z.size());
69 
70  std::random_device rd{};
71  std::mt19937 gen{rd()};
72 
73  realtype sigma;
74 
75  checkSigmaPositivity(sigma_y, "sigma_y");
76  checkSigmaPositivity(sigma_z, "sigma_z");
77 
78  for (int iy = 0; iy < nytrue; ++iy) {
79  for (int it = 0; it < nt(); ++it) {
80  sigma = sigma_y.size() == (unsigned) nytrue ? sigma_y.at(iy) : sigma_y.at(iy + nytrue * it);
81  std::normal_distribution<> e{0, sigma};
82  observedData.at(iy + nytrue * it) = rdata.y.at(iy + rdata.ny * it) + e(gen);
83  observedDataStdDev.at(iy + nytrue * it) = sigma;
84  }
85  }
86 
87  for (int iz = 0; iz < nztrue; ++iz) {
88  for (int ie = 0; ie < nmaxevent; ++ie) {
89  sigma = sigma_z.size() == (unsigned) nztrue ? sigma_z.at(iz) : sigma_z.at(iz + nztrue * ie);
90  std::normal_distribution<> e{0, sigma};
91  observedEvents.at(iz + rdata.nztrue * ie) = rdata.z.at(iz + rdata.nz * ie) + e(gen);
92  observedDataStdDev.at(iz + rdata.nztrue * ie) = sigma;
93  }
94  }
