77 void ForwardProblem::workForwardProblem() {
83  try {
84  solver->setup(&x, &dx, &sx, &sdx, model);
85  } catch (std::exception& ex) {
86  throw AmiException("AMICI setup failed:\n(%s)",ex.what());
87  } catch (...) {
88  throw AmiException("AMICI setup failed due to an unknown error");
89  }
90 
91  if(edata){
92  rdata->initializeObjectiveFunction();
93  }
94 
95  /* if preequilibration is necessary, start Newton solver */
96  if (solver->getNewtonPreequilibration() || (edata && !edata->fixedParametersPreequilibration.empty())) {
97  handlePreequilibration();
98  } else {
99  rdata->x0 = x.getVector();
100  if (solver->getSensitivityMethod() == SensitivityMethod::forward &&
101  solver->getSensitivityOrder() >= SensitivityOrder::first) {
102  for (int ix = 0; ix < model->nx; ix++) {
103  for (int ip = 0; ip < model->nplist(); ip++)
104  rdata->sx0[ip*model->nx + ix] = sx.at(ix,ip);
105  }
106  }
107  }
108 
109  int ncheck = 0; /* the number of (internal) checkpoints stored so far */
110 
111  /* perform presimulation if necessary */
112  if (edata && edata->t_presim > 0)
113  handlePresimulation(&ncheck);
114 
115  /* loop over timepoints */
116  for (int it = 0; it < model->nt(); it++) {
117  auto nextTimepoint = model->t(it);
118 
119  if (solver->getSensitivityMethod() == SensitivityMethod::forward &&
120  solver->getSensitivityOrder() >= SensitivityOrder::first) {
121  solver->setStopTime(nextTimepoint);
122  }
123 
124  if (nextTimepoint > model->t0()) {
125  while (t < nextTimepoint) {
126  if (model->nx == 0) {
127  t = nextTimepoint;
128  continue;
129  }
130 
131  if (std::isinf(nextTimepoint)) {
132  SteadystateProblem sstate = SteadystateProblem(&t,&x,&sx);
133  sstate.workSteadyStateProblem(rdata, solver, model, it);
134  } else {
135  int status;
136  if (solver->getSensitivityMethod() == SensitivityMethod::adjoint &&
137  solver->getSensitivityOrder() >= SensitivityOrder::first) {
138  status = solver->solveF(RCONST(nextTimepoint), &x, &dx,
139  &(t), AMICI_NORMAL, &ncheck);
140  } else {
141  status = solver->solve(RCONST(nextTimepoint), &x, &dx,
142  &(t), AMICI_NORMAL);
143  }
144 
145  if (status == AMICI_ILL_INPUT) {
146  /* clustering of roots => turn off rootfinding */
147  solver->turnOffRootFinding();
148  } else if (status == AMICI_ROOT_RETURN) {
149  handleEvent(&tlastroot, false);
150  }
151  }
152  }
153  }
154  handleDataPoint(it);
155  }
156 
157  /* fill events */
158  if (model->nz > 0) {
159  getEventOutput();
160  }
161 
162  // set likelihood
163  if (!edata) {
164  rdata->invalidateLLH();
165  }
166 
167  storeJacobianAndDerivativeInReturnData();
168 }

170 void ForwardProblem::handlePreequilibration()
171 {
172  // Are there dedicated condition preequilibration parameters provided?
173  bool overrideFixedParameters = edata && !edata->fixedParametersPreequilibration.empty();
174 
175  std::vector<realtype> originalFixedParameters; // to restore after pre-equilibration
176 
177  if(overrideFixedParameters) {
178  if(edata->fixedParametersPreequilibration.size() != (unsigned) model->nk())
179  throw AmiException("Number of fixed parameters (%d) in model does not match preequilibration parameters in ExpData (%zd).",
180  model->nk(), edata->fixedParametersPreequilibration.size());
181  originalFixedParameters = model->getFixedParameters();
182  model->setFixedParameters(edata->fixedParametersPreequilibration);
183  }
184 
185  // pre-equilibrate
186  SteadystateProblem sstate = SteadystateProblem(&t,&x,&sx);
187  sstate.workSteadyStateProblem(rdata, solver, model, -1);
188 
189  if(overrideFixedParameters) { // Restore
190  model->setFixedParameters(originalFixedParameters);
191  }
192 
193  updateAndReinitStatesAndSensitivities();

214 {
215  // Are there dedicated condition preequilibration parameters provided?
216  bool overrideFixedParameters = edata && !edata->fixedParametersPresimulation.empty();
217 
218  std::vector<realtype> originalFixedParameters; // to restore after pre-equilibration
219 
220  if(overrideFixedParameters) {
221  if(edata->fixedParametersPresimulation.size() != (unsigned) model->nk())
222  throw AmiException("Number of fixed parameters (%d) in model does not match presimulation parameters in ExpData (%zd).",

224  originalFixedParameters = model->getFixedParameters();

251 void ForwardProblem::handleEvent(realtype *tlastroot, const bool seflag) {
258  /* store heaviside information at event occurence */
259  model->froot(t, &x, &dx, rootvals.data());
260 
261  if (!seflag) {
262  solver->getRootInfo(rootsfound.data());
263  }
264 
265  if (iroot < model->nMaxEvent() * model->ne) {
266  std::copy(rootsfound.begin(), rootsfound.end(), &rootidx[iroot * model->ne]);
267  }
268 
269  rvaltmp = rootvals;
270 
271  if (!seflag) {
272  /* only extract in the first event fired */
273  if (solver->getSensitivityOrder() >= SensitivityOrder::first &&
274  solver->getSensitivityMethod() == SensitivityMethod::forward) {
275  solver->getSens(&(t), &sx);
276  }
277 
278  /* only check this in the first event fired, otherwise this will always
279  * be true */
280  if (t == *tlastroot) {
281  throw AmiException("AMICI is stuck in an event, as the initial"
282  "step-size after the event is too small. To fix "
283  "this, increase absolute and relative tolerances!");
284  }
285  *tlastroot = t;
286  }
287 
288  if(model->nz>0)
289  getEventOutput();
290 
291  /* if we need to do forward sensitivities later on we need to store the old
292  * x and the old xdot */
293  if (solver->getSensitivityOrder() >= SensitivityOrder::first) {
294  /* store x and xdot to compute jump in sensitivities */
295  x_old = x;
296  if (solver->getSensitivityMethod() == SensitivityMethod::forward) {
297  model->fxdot(t, &x, &dx, &xdot);
298  xdot_old = xdot;
299  dx_old = dx;
300 
301  /* compute event-time derivative only for primary events, we get
302  * into trouble with multiple simultaneously firing events here (but
303  * is this really well defined then?), in that case just use the
304  * last ie and hope for the best. */
305  if (!seflag) {
306  for (int ie = 0; ie < model->ne; ie++) {
307  if (rootsfound.at(ie) == 1) {
308  /* only consider transitions false -> true */
309  model->fstau(t, ie, &x, &sx);
310  }
311  }
312  }
313  } else if (solver->getSensitivityMethod() == SensitivityMethod::adjoint) {
314  /* store x to compute jump in discontinuity */
315  if (iroot < model->nMaxEvent() * model->ne) {
316  x_disc[iroot] = x;
317  xdot_disc[iroot] = xdot;
318  xdot_old_disc[iroot] = xdot_old;
319  }
320  }
321  }
322 
323  model->updateHeaviside(rootsfound);
324 
325  applyEventBolus();
326 
327  if (iroot < model->nMaxEvent() * model->ne) {
328  discs[iroot] = t;
329  ++iroot;
330  } else {
331  warnMsgIdAndTxt("AMICI:mex:TOO_MUCH_EVENT",
332  "Event was recorded but not reported as the number of "
333  "occured events exceeded (nmaxevents)*(number of "
334  "events in model definition)!");
335  solver->reInit(t, &x, &dx); /* reinitialise so that we can continue in peace */
336  return;
337  }
338 
339  if (solver->getSensitivityOrder() >= SensitivityOrder::first
340  && solver->getSensitivityMethod() == SensitivityMethod::forward) {
341  /* compute the new xdot */
342  model->fxdot(t, &x, &dx, &xdot);
343  applyEventSensiBolusFSA();
344  }
345 
346  int secondevent = 0;
347 
348  /* check whether we need to fire a secondary event */
349  model->froot(t, &x, &dx, rootvals.data());
350  for (int ie = 0; ie < model->ne; ie++) {
351  /* the same event should not trigger itself */
352  if (rootsfound.at(ie) == 0) {
353  /* check whether there was a zero-crossing */
354  if (0 > rvaltmp.at(ie) * rootvals.at(ie)) {
355  if (rvaltmp.at(ie) < rootvals.at(ie)) {
356  rootsfound.at(ie) = 1;
357  } else {
358  rootsfound.at(ie) = -1;
359  }
360  secondevent++;
361  } else {
362  rootsfound.at(ie) = 0;
363  }
364  } else {
365  /* don't fire the same event again */
366  rootsfound.at(ie) = 0;
367  }
368  }
369  /* fire the secondary event */
370  if (secondevent > 0) {
371  handleEvent(tlastroot, TRUE);
372  }
373 
374  /* only reinitialise in the first event fired */
375  if (!seflag) {
376  solver->reInit(t, &x, &dx);
377 
378  /* make time derivative consistent */
379  solver->calcIC(t, &x, &dx);
380 
381  if (solver->getSensitivityOrder() >= SensitivityOrder::first) {
382  if (solver->getSensitivityMethod() == SensitivityMethod::forward) {
383  solver->sensReInit(&sx, &sdx);
384  }
385  }
386  }
387 }

407 void ForwardProblem::getEventOutput() {
412  if (t == model->gett(model->nt() - 1,rdata)) {
413  // call from fillEvent at last timepoint
414  model->froot(t, &x, &dx, rootvals.data());
415  }
416 
417  /* EVENT OUTPUT */
418  for (int ie = 0; ie < model->ne; ie++) {
419  /* only look for roots of the rootfunction not discontinuities */
420  if (nroots.at(ie) >= model->nMaxEvent())
421  continue;
422 
423  /* only consider transitions false -> true or event filling */
424  if (rootsfound.at(ie) != 1 && t != model->gett(model->nt() - 1, rdata)) {
425  continue;
426  }
427 
428  model->fz(nroots.at(ie), ie, t, &x, rdata);
429 
430  if (edata) {
431  model->fsigmaz(t, ie, nroots.data(), rdata, edata);
432  model->fJz(nroots.at(ie), rdata, edata);
433 
434  if (t == model->gett(model->nt() - 1,rdata)) {
435  // call from fillEvent at last
436  // timepoint, add regularization
437  // based on rz
438  model->frz(nroots.at(ie), ie, t, &x, rdata);
439  model->fJrz(nroots.at(ie), rdata, edata);
440  }
441  }
442 
443  if (solver->getSensitivityOrder() >= SensitivityOrder::first) {
444  prepEventSensis(ie);
445  if (solver->getSensitivityMethod() == SensitivityMethod::forward) {
446  getEventSensisFSA(ie);
447  }
448  }
449 
450  nroots.at(ie)++;
451  }
452 
453  if (t == model->gett(model->nt() - 1, rdata)) {
454  // call from fillEvent at last timepoint
455  // loop until all events are filled
456  if(std::any_of(nroots.cbegin(), nroots.cend(), [&](int curNRoots){ return curNRoots < model->nMaxEvent(); }))
457  getEventOutput();
458  }
