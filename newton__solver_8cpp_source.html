363 void NewtonSolverIterative::linsolveSPBCG(int ntry,int nnewt, AmiVector *ns_delta) {
375  double rho;
376  double alpha;
377  double omega;
378  double res;
379 
380  xdot = *ns_delta;
381  xdot.minus();
382 
383  // Get the diagonal of the Jacobian for preconditioning
384  model->fJDiag(*t, &ns_Jdiag, 0.0, x, &dx);
385 
386  // Ensure positivity of entries in ns_Jdiag
387  ns_p.set(1.0);
388  N_VAbs(ns_Jdiag.getNVector(), ns_tmp.getNVector());
389  N_VCompare(1e-15, ns_tmp.getNVector(), ns_tmp.getNVector());
390  N_VLinearSum(-1.0, ns_tmp.getNVector(), 1.0, ns_p.getNVector(), ns_tmp.getNVector());
391  N_VLinearSum(1.0, ns_Jdiag.getNVector(), 1.0, ns_tmp.getNVector(), ns_Jdiag.getNVector());
392 
393  // Initialize for linear solve
394  ns_p.reset();
395  ns_v.reset();
396  ns_delta->reset();
397  ns_tmp.reset();
398  rho = 1.0;
399  omega = 1.0;
400  alpha = 1.0;
401 
402  // can be set to 0 at the moment
403  model->fJv(*t, x, &dx, &xdot, ns_delta, &ns_Jv, 0.0);
404 
405  // ns_r = xdot - ns_Jv;
406  N_VLinearSum(-1.0, ns_Jv.getNVector(), 1.0, xdot.getNVector(), ns_r.getNVector());
407  N_VDiv(ns_r.getNVector(), ns_Jdiag.getNVector(), ns_r.getNVector());
408  res = sqrt(N_VDotProd(ns_r.getNVector(), ns_r.getNVector()));
409  ns_rt = ns_r;
410 
411  for (int i_linstep = 0; i_linstep < maxlinsteps;
412  i_linstep++) {
413  // Compute factors
414  double rho1 = rho;
415  rho = N_VDotProd(ns_rt.getNVector(), ns_r.getNVector());
416  double beta = rho * alpha / (rho1 * omega);
417 
418  // ns_p = ns_r + beta * (ns_p - omega * ns_v);
419  N_VLinearSum(1.0, ns_p.getNVector(), -omega, ns_v.getNVector(), ns_p.getNVector());
420  N_VLinearSum(1.0, ns_r.getNVector(), beta, ns_p.getNVector(), ns_p.getNVector());
421 
422  // ns_v = J * ns_p
423  model->fJv(*t, x, &dx, &xdot, &ns_p, &ns_v, 0.0);
424  N_VDiv(ns_v.getNVector(), ns_Jdiag.getNVector(), ns_v.getNVector());
425 
426  // Compute factor
427  alpha = rho / N_VDotProd(ns_rt.getNVector(), ns_v.getNVector());
428 
429  // ns_h = ns_delta + alpha * ns_p;
430  N_VLinearSum(1.0, ns_delta->getNVector(), alpha, ns_p.getNVector(), ns_h.getNVector());
431  // ns_s = ns_r - alpha * ns_v;
432  N_VLinearSum(1.0, ns_r.getNVector(), -alpha, ns_v.getNVector(), ns_s.getNVector());
433 
434  // ns_t = J * ns_s
435  model->fJv(*t, x, &dx, &xdot, &ns_s, &ns_t, 0.0);
436  N_VDiv(ns_t.getNVector(), ns_Jdiag.getNVector(), ns_t.getNVector());
437 
438  // Compute factor
439  omega = N_VDotProd(ns_t.getNVector(), ns_s.getNVector()) / N_VDotProd(ns_t.getNVector(), ns_t.getNVector());
440 
441  // ns_delta = ns_h + omega * ns_s;
442  N_VLinearSum(1.0, ns_h.getNVector(), omega, ns_s.getNVector(), ns_delta->getNVector());
443  // ns_r = ns_s - omega * ns_t;
444  N_VLinearSum(1.0, ns_s.getNVector(), -omega, ns_t.getNVector(), ns_r.getNVector());
445 
446  // Compute the (unscaled) residual
447  N_VProd(ns_r.getNVector(), ns_Jdiag.getNVector(), ns_r.getNVector());
448  res = sqrt(N_VDotProd(ns_r.getNVector(), ns_r.getNVector()));
449 
450  // Test convergence
451  if (res < atol) {
452  // Write number of steps needed
453  rdata->newton_numlinsteps[(ntry - 1) * maxsteps +
454  nnewt] = i_linstep + 1;
455 
456  // Return success
457  return;
458  }
459 
460  // Scale back
461  N_VDiv(ns_r.getNVector(), ns_Jdiag.getNVector(), ns_r.getNVector());
462  }
463  throw NewtonFailure(AMICI_CONV_FAILURE, "linsolveSPBCG");
464 }
