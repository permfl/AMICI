518  species = self.sbml.getListOfSpecies()

520  self.speciesIndex = {species_element.getId(): species_index
521  for species_index, species_element
522  in enumerate(species)}

525  )
526  self.symbols['species']['name'] = [spec.getName() for spec in species]
527  self.speciesCompartment = sp.DenseMatrix(
528  [symbols(spec.getCompartment()) for spec in species]
529  )
530  self.constantSpecies = [species_element.getId()
531  for species_element in species
532  if species_element.getConstant()]
533  self.boundaryConditionSpecies = [
534  species_element.getId()
535  for species_element in species

537  ]
538  self.speciesHasOnlySubstanceUnits = [specie.getHasOnlySubstanceUnits()
539  for specie in species]

541  concentrations = [spec.getInitialConcentration() for spec in species]
542  amounts = [spec.getInitialAmount() for spec in species]

544  def getSpeciesInitial(index, conc):
545  if not math.isnan(conc):
546  return sp.sympify(conc)
547  if not math.isnan(amounts[index]):
548  return sp.sympify(amounts[index]) \
549  / self.speciesCompartment[index]
550  return self.symbols['species']['sym'][index]
551  self.speciesInitial = sp.DenseMatrix([getSpeciesInitial(index, conc)
552  for index, conc
553  in enumerate(concentrations)])

556  for initial_assignment in self.sbml.getListOfInitialAssignments():
557  if initial_assignment.getId() in species_ids:
558  index = species_ids\

560  initial_assignment.getId()
561  )
562  self.speciesInitial[index] = sp.sympify(
563  sbml.formulaToL3String(initial_assignment.getMath())
564  )
565 
566  # flatten initSpecies
567  while any([species in self.speciesInitial.free_symbols
568  for species in self.symbols['species']['sym']]):
569  self.speciesInitial = self.speciesInitial.subs(
570  self.symbols['species']['sym'],
571  self.speciesInitial
572  )
573 
574  if self.sbml.isSetConversionFactor():
575  conversion_factor = self.sbml.getConversionFactor()
576  else:
577  conversion_factor = 1.0
578 
579  self.speciesConversionFactor = sp.DenseMatrix([
580  sp.sympify(specie.getConversionFactor())
581  if specie.isSetConversionFactor()
582  else conversion_factor
583  for specie in species

681  def processReactions(self):
682  reactions = self.sbml.getListOfReactions()
683  self.n_reactions = len(reactions)
684 
685  # stoichiometric matrix
686  self.stoichiometricMatrix = sp.zeros(self.n_species, self.n_reactions)
687  self.fluxVector = sp.zeros(self.n_reactions, 1)
688  self.symbols['flux']['sym'] = sp.zeros(self.n_reactions, 1)
689 
690 
691  assignment_ids = [ass.getId()
692  for ass in self.sbml.getListOfInitialAssignments()]
693  rulevars = [rule.getVariable()
694  for rule in self.sbml.getListOfRules()
695  if rule.getFormula() != '']
696 
697  reaction_ids = [
698  reaction.getId() for reaction in reactions
699  if reaction.isSetId()
700  ]
701 
702  for reactionIndex, reaction in enumerate(reactions):
703 
704  for elementList, sign in [(reaction.getListOfReactants(), -1.0),
705  (reaction.getListOfProducts(), 1.0)]:
706  elements = {}
707  for index, element in enumerate(elementList):
708  # we need the index here as we might have multiple elements
709  # for the same species
710  elements[index] = {'species': element.getSpecies()}
711  if element.isSetId():
712  if element.getId() in assignment_ids:
713  assignment = self.sbml.getInitialAssignment(
714  element.getId()
715  )
716  elements[index]['stoichiometry'] = \
717  sp.sympify(assignment.getFormula())
718  # this is an initial assignment so we need to use
719  # initial conditions
720  elements[index]['stoichiometry'] = \
721  elements[index]['stoichiometry'].subs(
722  self.symbols['species']['sym'],
723  self.speciesInitial
724  )
725  elif element.getId() in rulevars:
726  elements[index]['stoichiometry'] = \
727  sp.sympify(element.getId())
728  else:
729  # dont put the symbol if it wont get replaced by a
730  # rule
731  elements[index]['stoichiometry'] = \
732  sp.sympify(element.getStoichiometry())
733  elif element.isSetStoichiometry():
734  elements[index]['stoichiometry'] = \
735  sp.sympify(element.getStoichiometry())
736  else:
737  elements[index]['stoichiometry'] = \
738  sp.sympify(1.0)
739 
740  for index in elements.keys():
741  if not (elements[index]['species'] in self.constantSpecies
742  or elements[index]['species'] in self.boundaryConditionSpecies):
743  specieIndex = self.speciesIndex[elements[index]['species']]
744  self.stoichiometricMatrix[specieIndex, reactionIndex] += sign * \
745  elements[index]['stoichiometry'] * self.speciesConversionFactor[specieIndex]/ \
746  self.speciesCompartment[specieIndex]
747 
748  # usage of formulaToL3String ensures that we get "time" as time symbol
749  math = sbml.formulaToL3String(reaction.getKineticLaw().getMath())
750  try:
751  symMath = sp.sympify(math)
752  except:
753  raise SBMLException('Kinetic law "' + math + '" contains an unsupported expression!')
754  for r in reactions:
755  for element in list(r.getListOfReactants()) + list(r.getListOfProducts()):
756  if element.isSetId() & element.isSetStoichiometry():
757  symMath = symMath.subs(sp.sympify(element.getId()),sp.sympify(element.getStoichiometry()))
758  self.fluxVector[reactionIndex] = symMath
759  self.symbols['flux']['sym'][reactionIndex] = sp.sympify('w' + str(reactionIndex))
760  if any([str(symbol) in reaction_ids
761  for symbol in self.fluxVector[reactionIndex].free_symbols]):
762  raise SBMLException('Kinetic laws involving reaction ids are currently not supported!')
